#快速幂

public class Solution {
    public double Power(double base, int exponent) {
        if(exponent<0){
            base = 1/base;
            exponent = 0-exponent;
        }
        return helper(base, exponent);
    }
    public double helper(double base, int exponent){
        if(exponent==0)
            return 1.0;
        if(exponent==1)
            return base;
        double ans = helper(base, exponent/2);
        if(exponent%2 == 0)
            return ans *ans;
        else{
            return ans*ans* base;
        }
    }
}

#非递归的快速幂

class Solution {
public:

    double Power(double b, int n) {
        if (n < 0) {
            b = 1 / b;
            n = -n;
        }
        double x = b; // 记录x^0, x^1, x^2 ...
        double ret = 1.0;
        while (n) {
            if (n&1) {
                ret *= x; // 二进制位数是1的，乘进答案。
            }
            x *= x;
            n >>= 1;
        }
        return ret;
    }
};

public class Solution {
    public void reOrderArray(int [] array) {
        int i = 0,j =0;
        while(i<array.length){
            if(array[i]%2==0){
                j = i+1;
                if(j==array.length)
                    return ;
                while(array[j]%2==0){
                    j++;
                    if(j==array.length)
                        return ;
                }
                int t = array[j];
                for(int xx = j;xx>i;xx--){
                    array[xx] = array[xx-1];
                }
                array[i] = t;
            }
            i++;
        }
    }
}

public class Solution {
    public ListNode ReverseList(ListNode head) {
        ListNode prev = null;
        ListNode cur = head;
        ListNode next = null;

        while(cur!=null){
            next = cur.next;
            cur.next = prev;
            prev = cur;
            cur = next ;
        }
        return prev;

    }
}

#合并排序链表

public class Solution {
    public ListNode Merge(ListNode list1,ListNode list2) {
        if(list1 == null)
            return list2;
        if(list2 == null)
            return list1;

        ListNode l = null;
        if(list1.val<=list2.val){
            l = list1;
            l.next = Merge(list1.next, list2);
        }

        if(list2.val<list1.val){
            l = list2;
            l.next = Merge(list1,list2.next);
        }
        return l;
    }
}

定义栈的数据结构，请在该类型中实现一个能够得到栈中所含最小元素的min函数（时间复杂度应为O（1））。

public class Solution {

    Stack<Integer> s = new Stack();
    Stack<Integer> mins = new Stack();
    public void push(int node) {
        s.push(node);
        if(mins.isEmpty()){
            mins.push(node);
        }else{
            if(mins.peek()<node){
                mins.push(mins.peek());
            }else{
                mins.push(node);
            }
        }

    }

    public void pop() {
        s.pop();
        mins.pop();
    }

    public int top() {
        return s.peek();
    }

    public int min() {
        return mins.peek();
    }
}

    public class Solution {
    public ArrayList<Integer> GetLeastNumbers_Solution(int [] input, int k) {
        ArrayList<Integer> ret = new ArrayList();
        if (k==0 || k > input.length) return ret;
        int right = input.length; int left = 0;
        while(left<right){
            int index = partition(input, left, right-1);
            if(index+1 == k){
                for(int i =0;i<k;i++){
                    ret.add(input[i]);
                }
                return ret;
            }
            if(index+1 < k){
                left = index+1;
            }
            else{
                right = index;
            }
        }
        return ret;
    }
    public int partition(int[] a, int low, int high){
        int temp = a[low];
        while(low<high){
            while(low<high && a[high]>=temp){
                high--;
            }
            a[low] = a[high];

            while(low<high && a[low]<=temp){
                low++;
            }
            a[high] = a[low];
        }
        a[low] = temp;
        return low;
    }
}

public static void mergeSort(int[] a, int i, int j){
        if(i ==j){
            return ;
        }
        int mid = (i+j)/2;
        mergeSort(a, i, mid);
        mergeSort(a, mid+1, j);

        merge(a, i, mid+1,j);
    }

    public static void merge(int[] a, int i, int mid, int j){
        int[] tleft = new int[mid-i];
        int[] tright = new int[j-mid+1];

        for(int x =i;x<mid;x++){
            tleft[x-i] = a[x];
        }
        for(int x =mid;x<=j;x++){
            tright[x-mid] = a[x];
        }
        int x=0,y=0,k=i;
        while(x<tleft.length&&y<tright.length){
            if (tleft[x] < tright[y]) {
                a[k] = tleft[x];

                x++;
                k++;
            } else {
                a[k] = tright[y];
                y++;
                k++;
            }

            x++;
            k++;
        }

            y++;
        }
    }
