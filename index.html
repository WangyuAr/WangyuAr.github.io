public class Solution {
    public double Power(double base, int exponent) {
        if(exponent<0){
            base = 1/base;
            exponent = 0-exponent;
        }
        return helper(base, exponent);
    }
    public double helper(double base, int exponent){
        if(exponent==0)
            return 1.0;
        if(exponent==1)
            return base;
        double ans = helper(base, exponent/2);
        if(exponent%2 == 0)
            return ans *ans;
        else{
            return ans*ans* base;
        }
    }
}

            b = 1 / b;
            n = -n;
        }
        double x = b; // 记录x^0, x^1, x^2 ...
        double ret = 1.0;
        while (n) {
            if (n&1) {
                ret *= x; // 二进制位数是1的，乘进答案。
            }
            x *= x;
            n >>= 1;
        }
        return ret;
    }

public class Solution {
    public ListNode Merge(ListNode list1,ListNode list2) {
        if(list1 == null)
            return list2;
        if(list2 == null)
            return list1;

        ListNode l = null;
        if(list1.val<=list2.val){
            l = list1;
            l.next = Merge(list1.next, list2);
        }

        if(list2.val<list1.val){
            l = list2;
            l.next = Merge(list1,list2.next);
        }
        return l;
    }
}

            mins.push(node);

            }else{
                mins.push(node);

    }

        return mins.peek();
    }

        int right = input.length; int left = 0;

            int index = partition(input, left, right-1);
            if(index+1 == k){
                for(int i =0;i<k;i++){
                    ret.add(input[i]);
                }
                return ret;
            }

                left = index+1;
            }
            else{
                right = index;
            }
        }
        return ret;
    }
    public int partition(int[] a, int low, int high){
        int temp = a[low];
        while(low<high){
            while(low<high && a[high]>=temp){
                high--;
            }
            a[low] = a[high];

            while(low<high && a[low]<=temp){
                low++;
            }
            a[high] = a[low];
        }
        a[low] = temp;
        return low;
    }
}

        if(i ==j){
            return ;
        }

        mergeSort(a, mid+1, j);

    public static void merge(int[] a, int i, int mid, int j){
        int[] tleft = new int[mid-i];
        int[] tright = new int[j-mid+1];

        for(int x =i;x<mid;x++){
            tleft[x-i] = a[x];
        }
        for(int x =mid;x<=j;x++){
            tright[x-mid] = a[x];
        }
        int x=0,y=0,k=i;
        while(x<tleft.length&&y<tright.length){
            if (tleft[x] < tright[y]) {
                a[k] = tleft[x];

                x++;
                k++;
            } else {
                a[k] = tright[y];
                y++;
                k++;
            }
        }

            a[k] = tleft[x];
            x++;
            k++;
        }
        while (y < tright.length) {
            a[k] = tright[y];
            k++;
            y++;
        }
    }

课程表 拓扑排序问题，判断是否存在环
先记录个用dfs做的：

思路： 依次遍历每个结点，遍历过程中遇到的结点状态值设为-1，遍历完成后的状态值设为1，初始状态值是0。

如果dfs过程中遇到状态值为-1的结点，说明肯定有环直接返回false.

如果dfs过程中遇到状态值为1的结点，说明被其他结点启动的dfs访问过了，肯定不存在环，返回true.

如果dfs过程中遇到状态值为0的结点，那就再以这个结点为起点dfs.

class Solution {
    int numCourses; int[][] prerequisites;
    int[] mem;
    public boolean canFinish(int numCourses, int[][] prerequisites) {
        this.numCourses = numCourses; this.prerequisites = prerequisites;
        this.mem = new int[numCourses];
        for (int i=0;i<numCourses;i++){
            if(mem[i] ==0 ){
                if(!dfs(i))
                    return false;
            }
        }
        return true;
    }
    public boolean dfs(int t){
        if(mem[t]==-1)
            return false;
        if(mem[t]==1)
            return true;
        mem[t] = -1;//当前访问结点

        for(int i =0;i<prerequisites.length;i++){
            int[] pt = prerequisites[i];
            if(pt[0]== t){
                if(!dfs(pt[1]))
                    return false;
            }
        }
        mem[t]=1;
        return true;
    }
}

给定一个二叉树，检查它是否是镜像对称的。

例如，二叉树 [1,2,2,3,4,4,3] 是对称的。

    1
   / \
  2   2
 / \ / \
3  4 4  3

但是下面这个 [1,2,2,null,3,null,3] 则不是镜像对称的:

    1
   / \
  2   2
   \   \
   3    3

递归解法：

class Solution {
    public boolean isSymmetric(TreeNode root) {
        if(root == null){
            return true;
        }
        return dfs(root,root);
    }
    public boolean dfs(TreeNode t1, TreeNode t2){
        if(t1 == null && t2 == null)
            return true;

        if(t1==null || t2==null)
            return false;
        return (t1.val==t2.val && dfs(t1.left,t2.right) && dfs(t2.left,t1.right));
    }
}

迭代解法：注意元素的添加顺序，不是按着每行顺序，是镜像顺序

public boolean isSymmetric(TreeNode root) {
    Queue<TreeNode> q = new LinkedList<>();
    q.add(root);
    q.add(root);
    while (!q.isEmpty()) {
        TreeNode t1 = q.poll();
        TreeNode t2 = q.poll();
        if (t1 == null && t2 == null) continue;
        if (t1 == null || t2 == null) return false;
        if (t1.val != t2.val) return false;
        q.add(t1.left);
        q.add(t2.right);
        q.add(t1.right);
        q.add(t2.left);
    }
    return true;
}

每天一道leetcode16（经典三指针法O（n^2））

發表於 2020-05-18 | 分類於 LeetCode

https://leetcode-cn.com/problems/3sum-closest/

给定一个包括 n 个整数的数组 nums 和 一个目标值 target。找出 nums 中的三个整数，使得它们的和与 target 最接近。返回这三个数的和。假定每组输入只存在唯一答案。

例如，给定数组 nums = [-1，2，1，-4], 和 target = 1.

与 target 最接近的三个数的和为 2. (-1 + 2 + 1 = 2).

思路： 经典三指针法，一个固定指针i,start= i+1，end = nums.length-1.

i从0到循环，对于每个i,用start和end遍历剩余数组。

int sum = nums[i]+nums[start]+nums[end];

如果sum>target,（数组有序）,那就end–;

如果sum<target,（数组有序）,那就start++;

别漏了sum==target的情况。

代码：

public int threeSumClosest(int[] nums, int target) {
        Arrays.sort(nums);
        int res = nums[0]+nums[1]+nums[2];
        for(int i=0;i<nums.length;i++){// 固定指针i
            int start = i+1;int end = nums.length-1;
            while(start<end){ // 两指针遍历
                int sum = nums[i]+nums[start]+nums[end];
                if( Math.abs(sum-target) < Math.abs(res-target)){
                    res = sum;
                }
                if(sum > target)
                    end --;
                if(sum < target)
                    start ++;

                if(sum== target){//别漏了
                    return sum;
                }
            }
        }
        return res;
    }

每天一道leetcode152（动态规划->乘积最大子数组）

發表於 2020-05-18 | 分類於 LeetCode

给你一个整数数组 nums ，请你找出数组中乘积最大的连续子数组（该子数组中至少包含一个数字），并返回该子数组所对应的乘积。

输入: [2,3,-2,4]
输出: 6
解释: 子数组 [2,3] 有最大乘积 6。

输入: [-2,0,-1]
输出: 0
解释: 结果不能为 2, 因为 [-2,-1] 不是子数组。

如果nums中不含有负数，那么动态规划的思路就很简单了。dpmax[i]代表从nums[0]到nums[i]的乘积最大值。状态转移方程为：

dp[i] = max{nums[i],dp[i-1]*nums[i]};

现在nums中包含负值，就设计两个dp数组，dpmax和dpmin。dpmax和dpmin意义相同，状态转移方程：

dpmax[i] = Math.max(Math.max(dpmax[i-1]*nums[i], nums[i]), dpmin[i-1]*nums[i]);

dpmin[i] = Math.min(Math.min(dpmin[i-1]*nums[i], nums[i]), dpmax[i-1]*nums[i]);

代码： 

public int maxProduct(int[] nums) {
    int[] dpmax = new int[nums.length];
    int[] dpmin = new int[nums.length];
    dpmax[0] = nums[0]; dpmin[0] = nums[0];
    int res = dpmax[0];
    for(int i=1;i<nums.length;i++){
        dpmax[i] = Math.max(Math.max(dpmax[i-1]*nums[i], nums[i]), dpmin[i-1]*nums[i]);
        dpmin[i] = Math.min(Math.min(dpmin[i-1]*nums[i], nums[i]), dpmax[i-1]*nums[i]);
        if(dpmax[i]>res)
            res = dpmax[i];
    }
    return res;
}

总结： dp求解的时候，首先要明确dp数组的含义。然后就可以推出状态转移方程了。

思路： 拓扑排序问题。先构建邻接表，然后可以通过DFS或者BFS来遍历。
（DFS很麻烦，有很多细节，推荐用BFS进行拓扑排序）。

BFS: 

记录每个结点的入度，将入度为0的结点塞入队列。
每次出队列，该结点相连的结点的入度减1，并判断减去1后入度是否为0， 为0就加入队列。
如果出队列的节点数等于总数说明可以拓扑排序，没有环，否则有环。

BFS代码：

class Solution {
    Map<Integer,List<Integer> > map;
    int numCourses;
    public int[] findOrder(int numCourses, int[][] prerequisites) {
        if (numCourses == 0) return new int[0];
        int [] res = new int[numCourses];
       // 用map构建邻接表
        Map<Integer,List<Integer>> map = new HashMap();

        int[] indegree = new int[numCourses];
        for (int i=0;i<prerequisites.length;i++){
            indegree[prerequisites[i][0]] = indegree[prerequisites[i][0]]+1; // 计算每个节点入度
            if(map.get(prerequisites[i][1])!=null){
                List<Integer> l = map.get(prerequisites[i][1]);
                l.add(prerequisites[i][0]);
                map.put(prerequisites[i][1], l);
            }else{
                List<Integer> l = new ArrayList();
                l.add(prerequisites[i][0]);
                map.put(prerequisites[i][1],l);
            }
        }

        Queue<Integer> queue = new LinkedList<>();
        //找入度为0的结点
        for (int i = 0; i < indegree.length; i++) {
            if (indegree[i] == 0) queue.offer(i);
        }
        int cnt = 0;
        while(!queue.isEmpty()){
            int curr = queue.poll();
            res[cnt++] = curr;
            List<Integer> l = map.get(curr);
            if(l!=null){
                for (int xx : l){
                    // 入度-1，并判断是否为0 
                    indegree[xx] = indegree[xx] -1;
                    if(indegree[xx] == 0){
                        queue.offer(xx);
                    }
                }
            }
        }
         if (cnt == numCourses) return res;
         return new int[0];
    }
}

每天一道leetcode124 (递归回溯最长路径)

發表於 2020-05-14 | 分類於 LeetCode

今天每日一题很简单，所以另外做了一道，Leecode124题。

给定一个非空二叉树，返回其最大路径和。
本题中，路径被定义为一条从树中任意节点出发，达到任意节点的序列。该路径至少包含一个节点，且不一定经过根节点。

输入: [1,2,3]

       1
      / \
     2   3

输出: 6

输入: [-10,9,20,null,null,15,7]

   -10
   / \
  9  20
    /  \
   15   7

输出: 42

思路：从根节点出发 深度优先遍历左右子树，得到左右子树的最大路径 left、right。

那么 max(root) = root.val + max(left,right);

如果必须经过根节点，那么结果就是max(root)。

现在可以不经过，这意味着我们要修改上面的函数，在每一步都检查哪种选择更好：是继续当前路径或者以当前节点作为最高节点计算新的路径。

  1. 初始化 max_sum 为最小可能的整数并调用函数 max_gain(node = root)。

  2. 实现 max_gain(node) 检查是继续旧路径还是开始新路径：

    • 边界情况：如果节点为空，那么最大权值是 0 。

    • 对该节点的所有孩子递归调用 max，计算从左右子树的最大权值：

      `left     = max     (max(node.left), 0)和right = max(max(node.right),0)。`

    • 检查是维护旧路径还是创建新路径。创建新路径的权值是：
      price_newpath = node.val + left + right，当新路径更好的时候更新 全局变量max_sum。

    • 对于递归返回的到当前节点的一条最大路径，计算结果为：
      node.val + max(left_gain, right_gain)

代码： 

class Solution {
    int ans = Integer.MIN_VALUE;
    public int maxPathSum(TreeNode root) {
        dfs(root);
        return ans;
    }
    public int dfs(TreeNode root){
        if(root==null)
            return 0;
        int left = Math.max(dfs(root.left),0);
        int right =  Math.max(dfs(root.right),0); 
        int curpath = root.val + Math.max(left,right);
        int newpath = root.val + left + right;
        ans = Math.max(ans,Math.max(curpath ,newpath));
        return curpath;
    }
}

今天的每日一题很简单，就是树的层次遍历。

给你一个二叉树，请你返回其按 层序遍历 得到的节点值。 （即逐层地，从左到右访问所有节点）。

示例：
二叉树：[3,9,20,null,null,15,7],

    3
   / \
  9  20
    /  \
   15   7
返回其层次遍历结果：

[
  [3],
  [9,20],
  [15,7]
]

思路： 利用队列，先进先出，注意记录当前层和当前层的节点数。

代码：

class Solution {
    public List<List<Integer>> levelOrder(TreeNode root) {
        List<List<Integer>> res = new ArrayList();

        if(root==null)
            return res;

        Queue<TreeNode> q = new LinkedList();
        q.add(root);
        int curlevel = 0;
        while(!q.isEmpty()){
            int level_len = q.size();//每一层的结点个数
            res.add(new ArrayList());
            for(int i =0;i<level_len;i++){
                TreeNode t = q.poll();
                res.get(curlevel).add(t.val);
                if(t.left!=null) // 每poll一个就add他的子节点进来
                    q.add(t.left);
                if(t.right!=null)
                    q.add(t.right);
            }
            curlevel++;

        }
        return res;
    }
}

每天一道leetcode236（递归回溯算法）

發表於 2020-05-12

今天的leetcode每日一题是第236题：二叉树的最近公共祖先

给定一个二叉树, 找到该树中两个指定节点的最近公共祖先。
https://leetcode-cn.com/problems/lowest-common-ancestor-of-a-binary-tree/

输入: root = [3,5,1,6,2,0,8,null,null,7,4], p = 5, q = 1
输出: 3
解释: 节点 5 和节点 1 的最近公共祖先是节点 3。

思路： 这题显然就是递归回溯，过程是

  1. 递归出口：当前结点是叶子节点
  2. 依次递归root的左子树 右子树，如果p、q分别在左右子树上那就是root,如果root是p或者q，并且另一个目标数字在root的左或者右子树上，也返回root

代码：（实质上可以看作是一个后序遍历）

class Solution {
    TreeNode p;
    TreeNode q;

    TreeNode res;

    public TreeNode lowestCommonAncestor(TreeNode root, TreeNode p, TreeNode q) {
        this.p = p;
        this.q = q;
        dfs(root);
        return res;
    }
    public boolean dfs(TreeNode root){
        if(root== null){ // 递归出口，是根节点
            return false;
        }
        boolean left = dfs(root.left);
        boolean right = dfs(root.right);
        if(left&&right || (root.val == p.val || root.val == q.val)&&(left||right))
            res = root;

        return (left||right || (root.val == p.val || root.val == q.val));
    }
}

还有一道递归回溯，leetcode周赛的一道题目。https://leetcode-cn.com/problems/minimum-time-to-collect-all-apples-in-a-tree/

思路： 也是后序遍历。
先看所有子节点是否有苹果；再看自身有没有苹果。

代码：

public class Leetcode5046 {

    public int minTime(int n, int[][] edges, List<Boolean> hasApple) {
        Map<Integer, List<Integer>> map = new HashMap();
        for (int i = 0; i < edges.length; i++) {
            int[] t = edges[i];
            List<Integer> l = null;
            if (map.get(t[0]) == null) {
                l = new ArrayList();
                l.add(t[1]);
            } else {
                l = map.get(t[0]);
                l.add(t[1]);
            }
            map.put(t[0], l);
        }
        int res = 0;
        List<Integer> children = map.get(0);
        for (int childIndex : children){
            res += dfs(n, map, hasApple, childIndex);
        }
        return res;
    }

    public int dfs(int n, Map<Integer, List<Integer>> map, List<Boolean> hasApple, int curNode) {
        List<Integer> l = map.get(curNode);
        if (l == null || l.size() == 0) {
            if (hasApple.get(curNode) == true) {
                return 2;
            }
            return 0;
        }
        int ans = 0;//注意！！！！！！！！！！！

        for (int i = 0; i < l.size(); i++) {
            ans = ans + dfs(n, map, hasApple, l.get(i));
        }

        if (ans > 0 ) {
            return ans + 2;
        } else if (hasApple.get(curNode) == true) {
            return 2;
        }else{
            return 0;
        }
    }
}
